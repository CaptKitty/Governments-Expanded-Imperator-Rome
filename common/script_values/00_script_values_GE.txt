GE_auxiliary_numbers_svalue = {
	value = 0
	scope:auxiliary_farm = {
		every_pops_in_province = {
			limit = {
				NOT = { pop_culture = country.culture }
				pop_type = freemen
			}
			add = 7
		}
		every_pops_in_province = {
			limit = {
				NOT = { pop_culture = country.culture }
				pop_type = freemen
				pop_happiness >= .75
			}
			add = 3
		}
		every_pops_in_province = {
			limit = {
				NOT = { pop_culture = country.culture }
				pop_type = freemen
				pop_happiness <= .50
			}
			subtract = 3
		}
		every_pops_in_province = {
			limit = {
				NOT = { pop_culture = country.culture }
				pop_type = freemen
				pop_happiness <= .25
			}
			subtract = 3
		}
	}
}
